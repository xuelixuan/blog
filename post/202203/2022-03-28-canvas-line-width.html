function draw() {
                let canvas = document.getElementById("canvas");
                if (canvas.getContext("2d")) {
                    let ctx = canvas.getContext("2d");
                    ctx.lineWidth = 20; // 线宽
                    ctx.beginPath();
                    ctx.moveTo(10, 10);
                    ctx.lineTo(30, 30);
                    ctx.stroke();
                }
            }
            window.onload = function () {
                draw();
            };
