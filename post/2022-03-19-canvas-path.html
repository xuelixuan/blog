function draw() {
                let canvas = document.getElementById("canvas");
                if (canvas.getContext("2d")) {
                    let ctx = canvas.getContext("2d");
                    ctx.beginPath();// 如果fillRect方法是画出矩形，那么beginPath方法就是为了画出自定义图形。
                    ctx.moveTo(75, 75); // 移动到某个坐标点
                    ctx.lineTo(100, 75);// 开始画一条线
                    ctx.lineTo(100, 25);// 画另外一条线
                    ctx.fill();// 对画出的自定义图形进行填充
                }
            }
            window.onload = function () {
                draw();
            };
